# 核心模块-path

basename() 荻取路径中基础名称

let path = require('path')
console.log(__filename)
console.log(path.basename(__filename))
console.log(path.basename(__filename,'.js'))
console.log(path.basename(__filename,'.css'))
console.log(path.basename('/a/b/c'))
console.log(path.basename('/a/b/c/'))
//D:\Desktop\Nodejs\Code\06-path. js
//06-path.js
//06-path
//06-path.js
//c
//c

dirname() 获取路径中目录名称

let path = require('path')
console.log(path.dirname(__filename))
console.log(path.dirname('/a/b/c'))
console.log(path.dirname('/a/b/c/'))
//D: \Desktop\Nodejs\Code 目录路径
// /a/b
// /a/b

extname() 获取路径中扩展名称

let path = require('path')
console.log(path.dirname(__filename))
console.log(path.dirname('/a/b/c'))
console.log(path.dirname('/a/b/index.html.css.js.scss'))
console.log(path.dirname('/a/b/index.html.css.js.'))
// .js
// (空)
// .scss
// .

parse() 解析路径

let path = require('path')
console.log(path.parse('/a/b/index.html'))
/*
{
  root: '/',
  dir: '/a/b',
  base: 'index.html',
  ext: '.html',
  name: 'index'
}
*/
console.log(path.parse('./a/b/index.html'))
/*
{
  root: '',
  dir: '/a/b',
  base: 'index.html',
  ext: '.html',
  name: 'index'
}
*/

format() 序列化路径

let path = require('path')
let obj = path.parse('./a/b/index.html')
console.log(path.format(obj))
// './a/b/index.html'

isAbsolute() 获取路径是否为绝对路径

let path = require('path')
console.log(path.isAbsolute('/foo'))
console.log(path.isAbsolute('///foo'))
console.log(path.isAbsolute('./foo'))
console.log(path.isAbsolute('.'))
console.log(path.isAbsolute(''))
console.log(path.isAbsolute('../foo'))
// true
// true
// false
// false
// false
// false

join() 拼接多个路径片段

let path = require('path')
console.log(path.join('a/b', 'c', 'index.html'))
console.log(path.join('/a/b', 'c', 'index.html'))
console.log(path.join('/a/b', 'c', '../', 'index.html'))
console.log(path.join('/a/b', 'c', './', 'index.html'))
// a/b/c/index.html
// /a/b/c/index.html
// /a/b/index.html
// /a/b/c/index.html

normalize(() 规范化路径

let path = require('path')
console.log(path.normalize(''))
console.log(path.normalize('a/b/c'))
console.log(path.normalize('a//////b..../c'))
console.log(path.normalize('a/\d/b../c'))
// .
// a/b/c/
// a/b..../c/
// a/d/b..../c/

resolve() 返回绝对路径

let path = require('path')
console.log(path.resolve())
console.log(path.resolve('index.html'))
// '/Users/wangkai/FrontEnd/yew-app-try'
// '/Users/wangkai/FrontEnd/yew-app-try/index.html'

# 二进制数据

文件的读写，网络的服务等使用的大多都是二进制数据。 在计算机眼里，无论是看到的听到的，亦或是我们写的代码其实都是"一串数字"，它也只认识这些数字。

# 流操作

可中断，配合管道技术可以实现数据分段传输

# Buffer

NodeJs全局变量 用于NodeJs平台下的二进制数据操作 不占用v8的内存空间，额外有一出内存空间来存放Buffer 还是由NodeJs来控制，垃圾回收也是由v8来控制 一般配合stream使用，充当数据缓冲区

# Buffer.alloc()

let buffer = Buffer.alloc(10)
console.log(buffer)
// Buffer[00,00,00,00,00,00,00,00,00,00]

# Buffer.from(data, encode)

可以接受三种数据类型：string、数组、buffer 默认encode是utf8格式 生成的buffer默认十六进制

Buffer.from('凯')
// <Buffer e5 87 af>

数组中不要传中文，不识别

Buffer.from(['凯'])
// <Buffer 00>

// 这里0207是87的八进制表示，其他则是16进制
Buffer.from([0xe5, 0207, 0xaf])
// <Buffer e5 87 af>
Buffer.from([0xe5, 0207, 0xaf]).toString()
// '凯'

TIP

let b = Buffer.alloc(3)
let b2 = Buffer.from(b)
// <Buffer 00 00 00>
// <Buffer 00 00 00>

b[0] = 1
// <Buffer 01 00 00>
// <Buffer 00 00 00>

有类似于"深拷贝"的意思，也就是Buffer.from只会利用alloc生成的内存空间"长度"，而不是简单的共享一个内存

# Buffer的实例方法

# fill

fill(value,offset, end = buf.length,encoding) 分别代表： 填充的值、偏移(起点)、结束位置、编码格式

let b = Buffer.alloc(6)
b.fill('哈')
// '哈'这个字符占3个字节，但我们声明了6个字节的空间
// fill会自动填充剩余的空间
console.log(b) // <Buffer e5 93 88 e5 93 88>
console.log(b.toString()) // 哈哈

不够字符的会被填充为0x00

let b = Buffer.alloc(6)
// 从1开始，4（不包含）结束
b.fill('哈', 1, 4)
console.log(b) // <Buffer 00 e5 93 88 00 00>
console.log(b.toString()) // 哈

# write

函数签名和fill一样，这里不重复了

let b = Buffer.alloc(6)
// 从1开始，3（包含）结束
b.write('1234', 1, 3)
console.log(b) // <Buffer 00 31 32 33 00 00>
console.log(b.toString()) // 123

# toString

let b = Buffer.alloc(6)
b.write('1234')
// 从1开始，3（不包含）结束
console.log(b.toString('1234', 1, 3))

# slice(废弃)，subarray

let b = Buffer.alloc(6)
b.write('1234')
let bb = b.subarray(2,4)
console.log(b)
// <Buffer 31 32 33 34 00 00>
console.log(bb)
// <Buffer 33 34>
console.log(bb.toString('utf8'))
// 34

可以是负数

let b = Buffer.alloc(6)
b.write('1234')
let bb = b.subarray(-3)
console.log(b)
// <Buffer 31 32 33 34 00 00>
console.log(bb)
// <Buffer 34 00 00>
console.log(bb.toString('utf8'))
// 4

let b1 = Buffer.alloc(6)

b2.copy(b1)

# FS模块

NodeJs内置核心模块，提供文件系统操作的API 用户对文件的操作权限： R: read 4 W: write 2 X: executor 1

一共10位，第一位：d或者-，d表示文件夹。后面九位和上面的图一一对应

flag表示对文件的操作方式
常见flag:
r: 可读 如果文件不存在，不会创建文件，会报错
w: 可写 如果不存在，会创建 s: 表示同步 +: 表示执行相反操作，如r+表示可读可写 x: 表示排它操作 a: 表示追加操作

fd是操作系统分配给被打开文件的标识
